// index.js const express = require('express'); const cors = require('cors'); const fetch = require('node-fetch'); const app = express(); app.use(express.json()); app.use(cors()); const TARGET_API_URL = 'http://187.198.168.170:3001/api/message'; // Replace with your actual API server app.post('/api/message', async (req, res) => { try { const response = await fetch(TARGET_API_URL, { method: 'POST', headers: { 'Content-Type': 'application/json' }, body: JSON.stringify(req.body), }); const data = await response.json(); res.json(data); } catch (error) { console.error('Error in proxy:', error); res.status(500).json({ error: 'Failed to forward request.' }); } }); const PORT = process.env.PORT || 3000; app.listen(PORT, () => console.log(`Proxy server running on port ${PORT}`));
